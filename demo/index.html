设置当前值，可设百分比

效果
代码 
                var s10 = new Slider({
                    $target: '#slider10',
                    orientation: 'horizontal',
                    type: 'max'
                });
                $('#J_btn_value').on('click', function () {
                    s10.setter('value', 30);
                });
                $('#J_btn_percent').on('click', function () {
                    s10.setter('value', '62%');
                });

析构组件

效果
代码 
                var s11 = new Slider({
                    $target: '#slider11',
                    orientation: 'horizontal',
                });
                $('#J_btn_display').on('click', function () {
                    var $self = $(this);
                    if ($self.html() === '隐藏组件') {
                        $self.html('显示组件');
                        s11.hide();
                    } else {
                        $self.html('隐藏组件');
                        s11.show();
                    }
                });
                $('#J_btn_destroy').on('click', function () {
                    var $self = $(this);
                    s11.destroy();
                    $self.attr('disabled', true);
                    $('#J_btn_display').attr('disabled', true);
                });
